def grabGlobalData(self):
    sql="SELECT countryName,shortCode,confirmed,cured,dead FROM Global WHERE date=\'%s\' ORDER BY confirmed DESC"
    #sql1=sql % self.today
    self.cursor.execute(sql % self.today)
    g_list=self.cursor.fetchall()
    g_Json=[]
    for g in g_list:
        gData={'name':g[0],'shortCode':g[1],'confirmed':g[2],'cured':g[3],'dead':g[4]}
        g_Json.append(gData)
    #print("g_Json=", g_Json)
    return g_Json

#从数据库提取国内数据
def grabChinaData(self):
    sql1="SELECT provinceName,confirmed,suspected,cured,dead,id FROM ChinaProvince WHERE date=\'%s\' ORDER BY confirmed DESC"
    sql2="SELECT cityName,confirmed,suspected,cured,dead,id FROM ChinaCity WHERE date=\'%s\' AND provinceName=\'%s\' ORDER BY confirmed DESC"
    self.cursor.execute(sql1 % self.today)
    p_list=self.cursor.fetchall()
    p_Json=[]
    for p in p_list:
        pData={'name':p[0],'confirmed':p[1],'suspected':p[2],'cured':p[3],'dead':p[4],'id':p[5]}
        cc=self.cursor.execute(sql2%(self.today,p[0]))
        if(cc!=0):
            c_list=self.cursor.fetchall()
            c_Json=[]
            for c in c_list:
                cData={'name':c[0],'confirmed':c[1],'suspected':c[2],'cured':c[3],'dead':c[4],'id':c[5]}
                c_Json.append(cData)
            if len(c_Json)!=0:
                pData.update({'children':c_Json})
        #else:
        #    print("pName=",p[0])
        p_Json.append(pData)
    #print("p_Json=",p_Json)
    return p_Json
